EXPECTED_BAKE_TIME = 40
def bake_time_remaining(time_in_oven):
    """
	returns remaining baketime
    takes the number of minutes the lasagne has been baking as an argument
    """
    time_left = EXPECTED_BAKE_TIME - time_in_oven
    return time_left

bake = bake_time_remaining(30)
print(bake)

def preparation_time_in_minutes(layers):
    """
	returns the perperation time
    takes the number of layers as an arguement
    """
    time = layers * 2
    return time
prep = preparation_time_in_minutes(2)
print(prep)

def elapsed_time_in_minutes(layers, time_in_oven):
    """
	Return elapsed cooking time.

    This function takes two numbers representing the number of layers & the time already spent 
    baking and calculates the total elapsed minutes spent cooking the
    """
    time_cooking = preparation_time_in_minutes(layers) + time_in_oven
    return time_cooking
elapse = elapsed_time_in_minutes(3, 20)
print(elapse)
